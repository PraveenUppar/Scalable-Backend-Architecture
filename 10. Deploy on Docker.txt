Docker 

// Without Docker Compose

// Backend image and running backend image in container
- docker build -t backend .
- docker images ls
- docker run --name todo-app -p 8080:8080 backend

// Run a mongoose container with its image
- docker run --name mongo_db -p 27107:27107 -d mongo

// Run a Redis container with its image
- docker run --name redis_db -p 6379:6379 -d redis
- docker exec -it redis bash
- redis-cli
- CONFIG SET requirepass password
- exit 

// Docker Network
- docker network create todo-network
- docker network ls
- docker network connect todo-network mongo_db 
- docker network connect todo-network redis_db 
- docker network inspect todo-network
- docker build -t backend .
- docker run --name todo-app -p 8080:8080 --network todo-network backend

// Docker Volumes
- docker run --name mongo -p 27107:27107 --network todo-network -v mongo-data:/data/db -d mongo 
- docker run --name redis -p 6379:6379 --network todo-network -v mongo-data:/data/db -d redis 

// Bind Mounts
- docker run --name todo-app -p 8080:8080 -v "$(pwd)":/app --network todo-network backend

// Pushing docker image to dockerhub
- docker build -t praveen/todo-app:latest

// With DockeR Compose
- docker compose up --build
